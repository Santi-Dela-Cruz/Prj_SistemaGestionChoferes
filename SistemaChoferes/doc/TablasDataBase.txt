CREATE TABLE `administrador` (
	`id` INT(10) NOT NULL AUTO_INCREMENT,
	`usuarioAdmin` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`nombreAdmin` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`apellidoAdmin` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`correoAdmin` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`contrasenaAdmin` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`fechaIngreso` DATE NOT NULL,
	`horaIngreso` TIME NOT NULL,
	`estadoAdmin` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`cargoAdmin` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`telefono` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	`direccion` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_spanish_ci',
	PRIMARY KEY (`id`) USING BTREE,
	UNIQUE INDEX `UNIQUE_CORREO` (`correoAdmin`, `contrasenaAdmin`, `usuarioAdmin`, `telefono`) USING BTREE
)
COLLATE='utf8mb3_spanish_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;


CREATE TABLE `choferes` (
	`id_Chofer` INT(10) NOT NULL AUTO_INCREMENT,
	`id_Cedula` VARCHAR(10) NOT NULL COLLATE 'utf8mb4_spanish_ci',
	`nombre` VARCHAR(100) NOT NULL COLLATE 'utf8mb4_unicode_ci',
	`apellido` VARCHAR(100) NOT NULL COLLATE 'utf8mb4_unicode_ci',
	`telefono` VARCHAR(10) NOT NULL COLLATE 'utf8mb4_spanish_ci',
	`direccion` VARCHAR(50) NOT NULL COLLATE 'utf8mb4_0900_ai_ci',
	`correo` VARCHAR(50) NOT NULL COLLATE 'utf8mb4_0900_ai_ci',
	`categoriaLicencia` VARCHAR(50) NOT NULL COLLATE 'utf8mb4_0900_ai_ci',
	`fechaVenciminetoLicencia` DATE NOT NULL,
	PRIMARY KEY (`id_Chofer`) USING BTREE,
	UNIQUE INDEX `UNIQUE KEY` (`id_Cedula`, `telefono`, `correo`) USING BTREE
)
COLLATE='utf8mb4_0900_ai_ci'
ENGINE=InnoDB
AUTO_INCREMENT=5
;

CREATE TABLE `huella` (
	`id_Huella` INT(10) NOT NULL AUTO_INCREMENT,
	`id_Codigo_Huella` VARCHAR(50) NOT NULL COLLATE 'utf8mb4_spanish_ci',
	`fecha_RegHuella` DATE NOT NULL,
	`id_Chofer` INT(10) NOT NULL,
	PRIMARY KEY (`id_Huella`) USING BTREE,
	UNIQUE INDEX `UK_CodigoHuella` (`id_Codigo_Huella`) USING BTREE,
	INDEX `FK_huella_choferes` (`id_Chofer`) USING BTREE,
	CONSTRAINT `FK_huella_choferes` FOREIGN KEY (`id_Chofer`) REFERENCES `choferes` (`id_Chofer`) ON UPDATE NO ACTION ON DELETE NO ACTION
)
COLLATE='utf8mb4_0900_ai_ci'
ENGINE=InnoDB
AUTO_INCREMENT=5
;

CREATE TABLE `registro_estados` (
